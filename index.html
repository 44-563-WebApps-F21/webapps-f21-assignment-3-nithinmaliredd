            vector> x(n*2);

            sort(x.begin(), x.end());
        
            int result = 0;
            int c = 0;
            for (int i = 0; i < n * 2; i++) {
                if (i > 0 && x[i].first > x[i-1].first && c > 0)
                    result += x[i].first - x[i-1].first;
                if (x[i].second)
                    c--;
                else
                    c++;
            }
            return result;
